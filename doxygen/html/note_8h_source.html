274 // In ARDUINO implementations, which to date is the largest use of this library,
275 // the Wire package is implemented in a broad variety of ways by different
276 // vendors. The default implementation has a mere 32-byte static I2C buffer,
277 // which means that the maximum to be transmitted in a single I/O (given our
278 // 2-byte serial-to-i2c protocol header) is 30 bytes. However, if we know
279 // the specific platform (such as STM32DUINO) we can relax this restriction.
280 #if defined(NOTE_I2C_MAX_DEFAULT)
281 // user is overriding it at compile time
282 #elif defined(ARDUINO_ARCH_STM32)
283 // we know that stm32duino dynamically allocates I/O buffer
284 #define NOTE_I2C_MAX_DEFAULT NOTE_I2C_MAX_MAX
285 #else
286 // default to what's known to be safe for all Arduino implementations
291 #define NOTE_I2C_MAX_DEFAULT 30
